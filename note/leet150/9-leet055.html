public boolean canJump(int[] nums) {
    boolean[] arr = new boolean[nums.length];
    arr[0] = true;
    for (int i = 0; i < nums.length; i++) {
        if (!arr[i]) {
            continue;
        }
        int range = nums[i];
        if (i + range >= nums.length - 1) {
            return true;
        }
        for (int j = i; j <= Math.min(i + range, nums.length - 1); j++) {
            arr[j] = true;
        }
    }
    return arr[nums.length - 1];
}
